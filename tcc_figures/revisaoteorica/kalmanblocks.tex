% The block diagram code is probably more verbose than necessary
\begin{tikzpicture}[auto, node distance=3cm,>=latex']
    % We start by placing the blocks
    \node [block, name=inputs] (inputs) {Entradas\\$\bm{z}_i$,$\bm{u}_i$};
    \node [block, below of=inputs, node distance = 2.5cm,fill=red!80.0!black!10, rounded corners] (predict) {Predição\\$\bm{x}_{est,i}=\bm{A}\cdot\bm{x}_{est,i-1}+\bm{B}\cdot\bm{u}_i$\\$\bm{P}_i=\bm{A}\cdot\bm{P}_{i-1}\cdot\bm{A}^T + \bm{Q}$};
    \node [block, below of=predict,name=update,fill=red!80.0!black!10,rounded corners] (update) {Correção\\$\bm{y}_i = \bm{z}_i - \bm{H}\cdot\bm{x}_{est,i}$\\$\bm{K}_i = \bm{P}_i\cdot\bm{H}^T\cdot(\bm{H}\cdot\bm{P}_i\cdot\bm{H}^T + \bm{R})^{-1}$\\$\bm{{x}}_{est,i} \leftarrow \bm{{x}}_{est,i} + \bm{K}_i\cdot\bm{y}_i$\\$\bm{P}_i \leftarrow (\bm{I}-\bm{K}_i\cdot\bm{H})\cdot\bm{P}_i$};
    \node [block,below of=update,fill=black!10] (outputs) {Saídas\\$\bm{x}_{est,i}$,$\bm{P}_i$};

    % Once the nodes are placed, connecting them is easy. 
    \node [output, right of=predict, node distance = 4cm] (outpredict) {};
    \node [input, left of=predict, node distance = 4cm] (inupdate) {};
    \node [output,left of=update, node distance = 4cm] (upoutput){};
    
    \draw [-] (predict) -- node {} (outpredict);
    \draw [->] (outpredict) |- (update);
    \draw [->] (update) -- (upoutput);
    \draw [-] (inupdate) -- (upoutput);
    \draw [->] (inupdate) -- node {$i\leftarrow i+1$} (predict);
    \draw [->] (inputs) -- (predict);
    \draw [->] (upoutput) |- (outputs);
    
    % \fill (inupdate) circle [radius=2pt];

\end{tikzpicture}
